import pandas as pd
import numpy as np

# הגדרות בסיס
file_name = 'data.xlsx'
sheets = ['בשרוני', 'ויה איטלי', 'הירוקה']
hours = ['10:00', '11:00', '12:00', '13:00', '14:00', '15:00']

def analyze_restaurant_complete(sheet_name):
    # 1. קריאת הנתונים מהלשונית
    df = pd.read_excel(file_name, sheet_name=sheet_name, header=None)
    
    # --- חלק א: ניתוח רווחיות ותפריט ---
    # דילוג על שורת הכותרת "מנה" (מתחילים משורה 1)
    menu_raw = df.iloc[1:6, 0:4].copy()
    menu_raw.columns = ['מנה', 'עלות', 'מחיר', 'זמן_הכנה']
    menu = menu_raw.dropna(subset=['מנה']).copy()
    
    # המרה למספרים וחישוב רווח גולמי
    for col in ['עלות', 'מחיר', 'זמן_הכנה']:
        menu[col] = pd.to_numeric(menu[col], errors='coerce')
    menu['רווח_למנה'] = menu['מחיר'] - menu['עלות']
    
    # --- חלק ב: ניתוח ביקושים ממוצעים ---
    # מציאת שורת תחילת הביקושים (חיפוש השעה 10:00)
    try:
        mask = df.astype(str).apply(lambda x: x.str.contains('10:00:00')).values
        row_idx, _ = np.where(mask)
        start_row = row_idx[0]
        
        # חילוץ נתוני הביקוש (עמודות B והלאה, שורות מתחת לשעות)
        demand_data = df.iloc[start_row + 1:, 1:].copy()
        demand_numeric = demand_data.apply(pd.to_numeric, errors='coerce')
        
        # חישוב ממוצע לכל עמודה (שעה של מנה)
        avg_per_col = demand_numeric.mean()
        
        demand_results = []
        col_idx = 0
        
        # חיבור בין המנות לביקושים שלהן
        for dish in menu['מנה'].tolist():
            dish_demand = {'מנה': dish}
            daily_total = 0
            
            for hour in hours:
                if col_idx < len(avg_per_col):
                    val = round(float(avg_per_col.iloc[col_idx]), 2)
                    dish_demand[hour] = val
                    daily_total += val
                    col_idx += 1
            
            dish_demand['סה"כ ביקוש יומי ממוצע'] = round(daily_total, 2)
            demand_results.append(dish_demand)
            
        df_demand_summary = pd.DataFrame(demand_results)
        
    except Exception as e:
        print(f"שגיאה בניתוח ביקושים למסעדת {sheet_name}: {e}")
        df_demand_summary = pd.DataFrame()

    # --- חלק ג: איחוד הנתונים לטבלה אחת מושלמת ---
    # מחברים את נתוני הרווח והזמן עם נתוני הביקוש
    final_summary = pd.merge(menu[['מנה', 'רווח_למנה', 'זמן_הכנה']], 
                             df_demand_summary, 
                             on='מנה')
    
    return final_summary

# הרצה על כל המסעדות והצגת התוצאות
for s in sheets:
    full_report = analyze_restaurant_complete(s)
    
    print(f"\n" + "="*60)
    print(f" דוח ניתוח נתונים סופי: מסעדת {s}")
    print("="*60)
    
    # הצגת הטבלה המאוחדת
    # סידרתי את העמודות שיהיה נוח לקרוא
    cols_order = ['מנה', 'רווח_למנה', 'זמן_הכנה', 'סה"כ ביקוש יומי ממוצע'] + hours
    display_df = full_report[cols_order]
    
    print(display_df.to_string(index=False))
    print("\n" + "-"*60)
